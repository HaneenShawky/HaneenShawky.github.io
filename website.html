Multithreaded Smart Elevator Simulator (C++)

This C++ implementation demonstrates a multithreaded elevator system with intelligent dispatching.

#include <iostream>
#include <thread>
#include <vector>
#include <queue>
#include <mutex>
#include <condition_variable>
#include <memory>
#include <chrono>
#include <algorithm>

using namespace std;
using Clock = chrono::steady_clock;

// A single elevator request
struct Request {
    int sourceFloor;
    int destFloor;
    Clock::time_point timestamp;
};

// Elevator class — one thread per elevator
class Elevator {
public:
    Elevator(int buildingId, int elevatorId)
        : buildingId(buildingId), elevatorId(elevatorId), currentFloor(1), running(true) {}

    void start() {
        thr = thread(&Elevator::run, this);
    }

    void assign(Request req) {
        lock_guard<mutex> lk(mtx);
        tasks.push(req);
        cv.notify_one();
    }

    void stop() {
        {
            lock_guard<mutex> lk(mtx);
            running = false;
            cv.notify_one();
        }
        thr.join();
    }

    int getCurrentFloor() {
        lock_guard<mutex> lk(mtx);
        return currentFloor;
    }

private:
    int buildingId;
    int elevatorId;
    int currentFloor;
    bool running;
    queue<Request> tasks;
    mutex mtx;
    condition_variable cv;
    thread thr;

    void run() {
        while (true) {
            unique_lock<mutex> lk(mtx);
            cv.wait(lk, [&] { return !tasks.empty() || !running; });
            if (!running && tasks.empty()) break;

            Request r = tasks.front(); tasks.pop();
            lk.unlock();

            auto start = Clock::now();
            moveTo(r.sourceFloor);
            log("Pick up at");
            moveTo(r.destFloor);
            log("Drop off at");
            auto end = Clock::now();
            auto ms = chrono::duration_cast<chrono::milliseconds>(end - r.timestamp).count();
            logDuration(ms);
        }
    }

    void moveTo(int target) {
        while (currentFloor != target) {
            this_thread::sleep_for(chrono::milliseconds(200));
            currentFloor += (currentFloor < target ? 1 : -1);
            log("Passing");
        }
    }

    void log(const char* action) {
        static mutex coutMtx;
        lock_guard<mutex> lk(coutMtx);
        cout << "[B" << buildingId << "-E" << elevatorId << "] "
            << action << " " << currentFloor << "\n";
    }

    void logDuration(long ms) {
        static mutex coutMtx;
        lock_guard<mutex> lk(coutMtx);
        cout << "[B" << buildingId << "-E" << elevatorId << "] "
            << "Request time: " << ms << " ms\n";
    }
};

// Building with dispatcher and elevators
class Building {
public:
    Building(int id, int numElev) : id(id), running(true) {
        for (int i = 0; i < numElev; i++)
            elevators.push_back(make_shared<Elevator>(id, i));
    }

    // Called in batch phase
    void collect(Request req) {
        lock_guard<mutex> lk(mtx);
        req.timestamp = Clock::now();
        reqQ.push(req);
    }

    // Start dispatcher + elevators
    void execute() {
        for (auto& e : elevators) e->start();
        disp = thread(&Building::dispatch, this);
    }

    // Stop everything
    void shutdown() {
        {
            lock_guard<mutex> lk(mtx);
            running = false;
            cv.notify_one();
        }
        disp.join();
        for (auto& e : elevators) e->stop();
    }

private:
    int id;
    bool running;
    queue<Request> reqQ;
    mutex mtx;
    condition_variable cv;
    vector<shared_ptr<Elevator>> elevators;
    thread disp;

    // Dispatcher thread fn
    void dispatch() {
        while (true) {
            unique_lock<mutex> lk(mtx);
            if (reqQ.empty() && !running) break;
            if (reqQ.empty()) {
                cv.wait_for(lk, chrono::milliseconds(100));
                continue;
            }
            Request r = reqQ.front(); reqQ.pop();
            lk.unlock();

            auto best = *min_element(elevators.begin(), elevators.end(),
                [&] (auto& a, auto& b) {
                    return abs(a->getCurrentFloor() - r.sourceFloor)
                        < abs(b->getCurrentFloor() - r.sourceFloor);
                });
            best->assign(r);
        }
    }
};

Quotes Mobile App (Flutter)

This Flutter implementation demonstrates a mobile app for displaying inspirational quotes with secure storage handling.

import 'dart:async';
import 'dart:collection';
import 'package:flutter/material.dart';
import 'dart:math';

void main() {
  runApp(ElevatorApp());
}

class ElevatorApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: ElevatorHome(),
    );
  }
}

class Request {
  final int source;
  final int dest;
  final DateTime timestamp;
  Request(this.source, this.dest) : timestamp = DateTime.now();
}

class Elevator {
  final int buildingId;
  final int elevatorId;
  int currentFloor = 1;
  final Queue<Request> tasks = Queue();
  bool running = true;
  final ValueNotifier<String> logNotifier;

  Elevator(this.buildingId, this.elevator极, this.logNotifier);

  void assign(Request r) {
    tasks.add(r);
    if (!_isRunning) run();
  }

  bool _isRunning = false;

  void run() async {
    _isRunning = true;
    while (tasks.isNotEmpty) {
      Request r = tasks.removeFirst();
      await moveTo(r.source);
      log("Pick up at");
      await moveTo(r.dest);
      log("Drop off at");
      final ms = DateTime.now().difference(r.timestamp).inMilliseconds;
      log("Request time: $ms ms");
    }
    _isRunning = false;
  }

  Future<void> moveTo(int target) async {
    while (currentFloor != target) {
      await Future.delayed(Duration(milliseconds: 200));
      currentFloor += (currentFloor < target ? 1 : -1);
      log("Passing");
    }
  }

  void log(String action) {
    logNotifier.value += "[B$buildingId-E$elevatorId] $action $currentFloor\n";
  }
}

class Building {
  final int id;
  final List<Elevator> elevators;
  final Queue<Request> reqQ = Queue();
  final ValueNotifier<String> logNotifier;

  Building(this.id, int numElevators, this.logNotifier)
      : elevators = List.generate(
      numElevators, (i) => Elevator(id, i, logNotifier));

  void collect(Request req) {
    reqQ.add(req);
  }

  void execute() async {
    while (reqQ.isNotEmpty) {
      Request r = reqQ.removeFirst();
      Elevator best = elevators.reduce((a, b) =>
      (r.source - a.currentFloor).abs() < (r.source - b.currentFloor).abs()
          ? a
          : b);
      best.assign(r);
      await Future.delayed(Duration(milliseconds: 100));
    }
  }
}

class ElevatorHome extends StatefulWidget {
  @override
  _ElevatorHomeState createState() => _ElevatorHomeState();
}

class _ElevatorHomeState extends State<ElevatorHome> {
  final logNotifier = ValueNotifier<String>("");
  final List<Building> buildings = [];
  final TextEditingController buildingController = TextEditingController();
  final TextEditingController floorController = TextEditingController();
  int numBuildings = 0;
  int floorsPerBuilding = 0;

  void setupBuildings() {
    numBuildings = int.tryParse(buildingController.text) ?? 1;
    floorsPerBuilding = int.tryParse(floorController.text) ?? 10;
    buildings.clear();
    for (int i = 0; i < numBuildings; i++) {
      buildings.add(Building(i, 1, logNotifier));
    }
  }

  void simulate() async {
    final simStart = DateTime.now();
    for (var b in buildings) {
      b.execute();
    }
    await Future.delayed(Duration(seconds: 2));
    final simEnd = DateTime.now();
    final totalMs = simEnd.difference(simStart).inMilliseconds;
    logNotifier.value += "Total simulation time: $totalMs ms\n";
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Elevator Simulation")),
      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          children: [
            TextField(
              controller: buildingController,
              decoration: InputDecoration(labelText: 'Number of Buildings'),
              keyboardType: TextInputType.number,
            ),
            TextField(
              controller: floorController,
              decoration: InputDecoration(labelText: 'Floors per Building'),
              keyboardType: TextInputType.number,
            ),
            ElevatedButton(
              onPressed: setupBuildings,
              child: Text('Initialize'),
            ),
            ElevatedButton(
              onPressed: () {
                // Add sample requests

                buildings[0].collect(Request(3, 1));
                simulate();
              },
              child: Text("Start Simulation"),
            ),
            Expanded(
              child: ValueListenableBuilder<String>(
                valueListenable: logNotifier,
                builder: (context, log, _) => SingleChildScrollView(
                  child: Text(log),
                ),
              ),
            )
          ],
        ),
      ),
    );
  }
}
